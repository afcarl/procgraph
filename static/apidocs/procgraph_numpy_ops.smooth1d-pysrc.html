10      """ 
11      Smooth the data using a window with requested size. 
12       
13      This method is based on the convolution of a scaled window with the signal. 
14      The signal is prepared by introducing reflected copies of the signal  
15      (with the window size) in both ends so that transient parts are minimized 
16      in the begining and end part of the output signal. 
17       
18      ``window`` must be one of  'flat', 'hanning', 'hamming', 'bartlett',  
19      'blackman'. 
20      A flat window will produce a moving average smoothing. 
21   
22       
23      :param x: the input signal  
24      :param window_len: the dimension of the smoothing window;  an odd integer 
25      :param window: the type of window from  
26      :return: smoothed: the smoothed signal 
27           
28      example: :: 
29   
30          t=linspace(-2,2,0.1) 
31          x=sin(t)+randn(len(t))*0.1 
32          y=smooth(x) 
33       
34      see also:  
35       
36      numpy.hanning, numpy.hamming, numpy.bartlett, numpy.blackman,  
37      numpy.convolve 
38      scipy.signal.lfilter 
39    
40      TODO: the window parameter could be the window itself if an  
41            array instead of a string    
42      """ 
43   
44      if x.ndim != 1: 
45          raise ValueError("smooth only accepts 1 dimension arrays.") 
46   
47      if x.size < window_len: 
48          raise ValueError("Input vector needs to be bigger than window size.") 
49   
50      if window_len < 3: 
51          return x 
52   
53      if not window in ['flat', 'hanning', 'hamming', 'bartlett', 'blackman']: 
54          raise ValueError( 
55                          "Window %r is not one of 'flat', 'hanning', 'hamming'," 
56                           "'bartlett', 'blackman'." % window) 
57   
58      s = numpy.r_[2 * x[0] - x[window_len:1:-1], 
59                   x, 
60                   2 * x[-1] - x[-1:-window_len:-1]] 
61      #print(len(s)) 
62      if window == 'flat': #moving average 
63          w = numpy.ones(window_len, 'd') 
64      else: 
65          w = eval('numpy.' + window + '(window_len)') 
66   
67      y = numpy.convolve(w / w.sum(), s, mode='same') 
68      return y[window_len - 1:-window_len + 1]
